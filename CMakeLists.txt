# Default setup
cmake_minimum_required(VERSION 3.1)
set(ASTEROID_VERSION_MAJOR 0)
set(ASTEROID_VERSION_MINOR 0)
set(ASTEROID_VERSION_PATCH 1)
set(ASTEROID_VERSION ${ASTEROID_VERSION_MAJOR}.${ASTEROID_VERSION_MINOR}.${ASTEROID_VERSION_PATCH})

project(ASTEROID)

# for multi arch LIBDIR support (requires cmake>=2.8.8)
INCLUDE(GNUInstallDirs)

# Set output folder
set(OUTPUT_DIR "${PROJECT_SOURCE_DIR}/bin")
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${OUTPUT_DIR})
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${OUTPUT_DIR})
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${OUTPUT_DIR})

# Avoid the extra "Debug", "Release" directories
foreach(OPTIONING ${CMAKE_CONFIGURATION_TYPES})
    string( TOUPPER ${OPTIONING} OPTIONING )
    set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_${OPTIONING} ${OUTPUT_DIR})
    set(CMAKE_LIBRARY_OUTPUT_DIRECTORY_${OPTIONING} ${OUTPUT_DIR})
    set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY_${OPTIONING} ${OUTPUT_DIR})
endforeach()

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Define the executable
add_executable(ASTEROID ${PROJECT_SOURCE_DIR}/src/Main.cpp)
